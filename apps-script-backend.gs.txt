/* ============================================================
   APPS SCRIPT BACKEND — free, no plugin, reused across all
   100+ tests. You set this up ONCE.

   WHAT IT DOES
   - Lets quiz-engine.html check "has this email already
     submitted THIS test?" before allowing a start (no retakes).
   - Receives each finished test, logs a row to a Google Sheet,
     and writes a formatted Google Doc into a Drive folder that
     only you own — the test-taker never gets access to it, so
     it's effectively non-editable by them.
   - Optionally converts that Doc to an actual .docx file.

   ============================================================
   SETUP (about 10 minutes, once)
   ============================================================
   1. Go to sheets.new — create a blank Google Sheet. This will
      be your submissions log. Copy its ID from the URL:
      https://docs.google.com/spreadsheets/d/THIS_PART_HERE/edit

   2. Go to drive.google.com — create a folder called e.g.
      "Test Results". Open it, copy its ID from the URL:
      https://drive.google.com/drive/folders/THIS_PART_HERE

   3. Go to script.google.com/create — a new Apps Script project.
      Delete the placeholder code and paste in everything below
      (from "const SHEET_ID" down to the end of the file).

   4. Paste your Sheet ID and Folder ID into the two constants
      right below this comment block.

   5. Click Deploy > New deployment > gear icon > Web app.
        - Execute as: Me
        - Who has access: Anyone
      Click Deploy, authorize when prompted (it'll warn you it's
      an unverified app — that's expected, it's your own script;
      click Advanced > Go to [project name] > Allow).

   6. Copy the Web app URL it gives you (ends in /exec). Paste
      that into ENGINE_CONFIG.appsScriptUrl in quiz-engine.html.
      That's the only line in the engine file you ever touch,
      and only once — every test after that reuses it.

   7. Re-run "Deploy > Manage deployments" if you ever edit this
      script later — you need to create a new version for
      changes to take effect on the live URL.
   ============================================================ */

const SHEET_ID = "PASTE_YOUR_SHEET_ID_HERE";
const FOLDER_ID = "PASTE_YOUR_FOLDER_ID_HERE";

// Set to false if you'd rather keep results as Google Docs
// instead of converting to .docx (Google Docs are just as
// private/non-editable to test-takers, and the conversion step
// below relies on an unofficial export endpoint that could
// break if Google changes it — this flag is your safety valve).
const CONVERT_TO_DOCX = true;


function doGet(e) {
  const action = e.parameter.action;
  if (action === 'check') {
    const email = (e.parameter.email || '').toLowerCase().trim();
    const testCode = (e.parameter.testCode || '').trim();
    const already = hasAlreadySubmitted(email, testCode);
    return jsonOut({ alreadySubmitted: already });
  }
  return jsonOut({ error: 'unknown action' });
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    logSubmission(payload);
    const fileUrl = buildReportDoc(payload);
    return jsonOut({ ok: true, fileUrl: fileUrl });
  } catch (err) {
    return jsonOut({ ok: false, reason: String(err) });
  }
}

function getLogSheet() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sheet = ss.getSheetByName('Submissions');
  if (!sheet) {
    sheet = ss.insertSheet('Submissions');
    sheet.appendRow(['Timestamp', 'Email', 'Test Code', 'Name', 'Score', 'Possible', 'Auto-submitted (timeout)']);
  }
  return sheet;
}

function hasAlreadySubmitted(email, testCode) {
  const sheet = getLogSheet();
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    const rowEmail = String(data[i][1] || '').toLowerCase().trim();
    const rowTest = String(data[i][2] || '').trim();
    if (rowEmail === email && rowTest === testCode) return true;
  }
  return false;
}

function logSubmission(payload) {
  const sheet = getLogSheet();
  sheet.appendRow([
    new Date(),
    payload.takerEmail,
    payload.testCode,
    payload.takerName,
    payload.score.earned,
    payload.score.possible,
    payload.autoSubmittedOnTimeout
  ]);
}

function buildReportDoc(payload) {
  const folder = DriveApp.getFolderById(FOLDER_ID);
  const doc = DocumentApp.create(payload.testCode + ' - ' + payload.takerName);
  const body = doc.getBody();

  body.appendParagraph(payload.testTitle).setHeading(DocumentApp.ParagraphHeading.TITLE);
  body.appendParagraph('Name: ' + payload.takerName);
  body.appendParagraph('Email: ' + payload.takerEmail);
  body.appendParagraph('Test code: ' + payload.testCode);
  body.appendParagraph('Submitted: ' + payload.submittedAt + (payload.autoSubmittedOnTimeout ? '  (auto-submitted — time ran out)' : ''));
  if (payload.score.gradableQuestionCount > 0) {
    body.appendParagraph('Score: ' + payload.score.earned + ' / ' + payload.score.possible);
  }
  body.appendHorizontalRule();

  payload.answers.forEach(a => {
    body.appendParagraph('Q' + a.number + '. ' + a.prompt).setHeading(DocumentApp.ParagraphHeading.HEADING3);
    body.appendParagraph('Answer: ' + (a.answerGiven || '(no answer given)'));
    if (a.type === 'mc') {
      body.appendParagraph((a.correct ? 'Correct' : 'Incorrect') + ' — correct answer: ' + a.correctAnswer);
    } else {
      body.appendParagraph('(free-text answer — review manually)');
    }
  });

  doc.saveAndClose();
  const docFile = DriveApp.getFileById(doc.getId());

  if (!CONVERT_TO_DOCX) {
    docFile.moveTo(folder);
    return docFile.getUrl();
  }

  try {
    const url = 'https://docs.google.com/feeds/download/documents/export/Export?id=' + doc.getId() + '&exportFormat=docx';
    const token = ScriptApp.getOAuthToken();
    const response = UrlFetchApp.fetch(url, { headers: { Authorization: 'Bearer ' + token } });
    const docxBlob = response.getBlob().setName(payload.testCode + ' - ' + payload.takerName + '.docx');
    const docxFile = folder.createFile(docxBlob);
    docFile.setTrashed(true); // remove the intermediate Google Doc, keep only the .docx
    return docxFile.getUrl();
  } catch (err) {
    // Conversion failed (e.g. Google changed the export endpoint) —
    // fall back to keeping the Google Doc so nothing is lost.
    docFile.moveTo(folder);
    return docFile.getUrl();
  }
}

function jsonOut(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
